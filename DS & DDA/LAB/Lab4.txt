01. Procedure: UpdateCourse

CREATE OR REPLACE PROCEDURE UpdateCourse (name_in IN VARCHAR2) 
IS
    v_course_number NUMBER;
BEGIN
    BEGIN
        SELECT course_number INTO v_course_number 
        FROM courses 
        WHERE course_name = name_in;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            v_course_number := 10000; -- Default value [cite: 18]
    END;

    INSERT INTO student_courses (course_name, course_number)
    VALUES (name_in, v_course_number);
END UpdateCourse;
/

DROP PROCEDURE UpdateCourse;    

02. Trigger: Salary Difference

CREATE OR REPLACE TRIGGER TRG_CUSTOMER_SALARY_DIFF
BEFORE INSERT OR UPDATE OR DELETE ON CUSTOMERS
FOR EACH ROW
DECLARE
    sal_diff NUMBER;
BEGIN
    IF UPDATING THEN
        sal_diff := :NEW.Salary - :OLD.Salary;
        DBMS_OUTPUT.PUT_LINE('Salary Difference: ' || sal_diff);
    ELSIF INSERTING THEN
        DBMS_OUTPUT.PUT_LINE('New Salary: ' || :NEW.Salary);
    ELSIF DELETING THEN
        DBMS_OUTPUT.PUT_LINE('Deleted Salary: ' || :OLD.Salary);
    END IF;
END;
/

DECLARE
    CURSOR emp_cur IS SELECT Name, hire_date FROM employees;
    v_years NUMBER;
BEGIN
    FOR r IN emp_cur LOOP
        v_years := FLOOR(MONTHS_BETWEEN(SYSDATE, r.hire_date)/12);
        DBMS_OUTPUT.PUT_LINE('Employee: ' || r.Name || ' Experience: ' || v_years || ' years');
    END LOOP;
END;

DECLARE
    CURSOR c_dept50 IS 
        SELECT Salary FROM employees WHERE deptid = 50 
        FOR UPDATE;
BEGIN
    FOR r IN c_dept50 LOOP
        UPDATE employees SET Salary = Salary * 1.1 
        WHERE CURRENT OF c_dept50;
    END LOOP;
END;
